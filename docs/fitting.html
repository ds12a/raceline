 12def g(t: float, x, q, u, spline_c: BSpline, spline_l: BSpline, spline_r: BSpline):
 13    """
 14    Calculates path error g(t) at a specific t.
 15    For track fitting, used both for computing the symbolic cost in CasADi
 16    for optimization and evaluating the numeric generated cost for
 17    hp-refinement.
 18    g(t)
 19
 20    Args:
 21        t (float): Arc length parameter
 22        x (CasADi Expression | np.ndarray): Vector containing [x, y, z]
 23        q (CasADi Expression | np.ndarray): Vector containing [θ, μ, Φ, n_l, n_r]
 24        u (CasADi Expression | np.ndarray): The control vector of second derivatives. u = ddq
 25        spline_c (BSpline): Scipy BSpline for center line
 26        spline_l (BSpline): Scipy BSpline for left boundary
 27        spline_r (BSpline): Scipy BSpline for right boundary
 28
 29    Returns:
 30        CasADi Expression | float: g(.)
 31    """
 32
 33    # ==================== Start Defining sub-error functions ====================
 34    def e(
 35        t: float,
 36        x,
 37        q,
 38        spline_c: BSpline,
 39        spline_l: BSpline,
 40        spline_r: BSpline,
 41        w_c: float = 1e-3,
 42        w_l: float = 1e-3,
 43        w_r: float = 1e-3,
 44    ):
 45        """
 46        Computes tracking error
 47        e = w_c ||x - c_spline(t)||^2 + w_l ||b_l - l_spline(t)||^2 + w_r ||b_r - r_spline(t)||^2
 48        where
 49        b_l = x + n * n_l
 50        b_r = x + n * n_r
 51        n = second column of Euler Rotation Matrix
 52
 53        Args:
 54            t (float): Arc length parameter
 55            x (CasADi Expression | np.ndarray): Vector containing [x, y, z]
 56            q (CasADi Expression | np.ndarray): Vector containing [theta, mu, phi]
 57            spline_c (BSpline): Scipy BSpline for center line
 58            spline_l (BSpline): Scipy BSpline for left boundary
 59            spline_r (BSpline): Scipy BSpline for right boundary
 60            w_c (float, optional): Defaults to 1e-3.
 61            w_l (float, optional): Defaults to 1e-3.
 62            w_r (float, optional): Defaults to 1e-3.
 63
 64        Returns:
 65            CasADi Expression | np.ndarray: Tracking error
 66        """
 67
 68        theta = q[0]
 69        mu = q[1]
 70        phi = q[2]
 71        n_l = q[3]
 72        n_r = q[4]
 73
 74        n = vertcat(
 75            cos(theta) * sin(mu) * sin(phi) - sin(theta) * cos(phi),
 76            sin(theta) * sin(mu) * sin(phi) + cos(theta) * cos(phi),
 77            cos(mu) * sin(phi),
 78        )
 79
 80        b_l = x.T + n * n_l
 81        b_r = x.T + n * n_r
 82
 83        x_c, y_c, z_c = splev(t, spline_c)
 84        x_l, y_l, z_l = splev(t, spline_l)
 85        x_r, y_r, z_r = splev(t, spline_r)
 86
 87        return (
 88            w_c * ((x[0] - x_c) ** 2 + (x[1] - y_c) ** 2 + (x[2] - z_c) ** 2)
 89            + w_l * ((b_l[0] - x_l) ** 2 + (b_l[1] - y_l) ** 2 + (b_l[2] - z_l) ** 2)
 90            + w_r * ((b_r[0] - x_r) ** 2 + (b_r[1] - y_r) ** 2 + (b_r[2] - z_r) ** 2)
 91        )
 92
 93    def r_c(u, w_theta=3e3, w_mu=1e9, w_phi=1e9):
 94        """
 95        Computes the error term that penalizes track curvature
 96        r_c = w_theta * dd_theta^2 + w_mu * dd_mu^2 + w_phi * dd_phi^2
 97
 98        Args:
 99            u (CasADi Expression | np.ndarray): The control vector of second derivatives. u = ddq
100            w_theta (float, optional): Defaults to 3e3.
101            w_mu (float, optional): Defaults to 1e9.
102            w_phi (float, optional): Defaults to 1e9.
103
104        Returns:
105            CasADi Expression | np.ndarray: r_c
106        """
107        return w_theta * u[0] ** 2 + w_mu * u[1] ** 2 + w_phi * u[2] ** 2
108
109    def r_w(u, w_n_l=1e5, w_n_r=1e5):
110        """
111        Computes the error term that penalizes track boundary noise
112        r_w = w_n_l * dd_n_l^2 + w_n_r * dd_n_r^2
113
114        Args:
115            u (CasADi Expression | np.ndarray): The control vector of second derivatives. u = ddq
116            w_n_l (_type_, optional): Defaults to 1e2.
117            w_n_r (_type_, optional): Defaults to 1e2.
118
119        Returns:
120            CasADi Expression | np.ndarray: r_w
121        """
122        return w_n_l * u[3] ** 2 + w_n_r * u[4] ** 2
123
124    # ==================== End Defining sub-error functions ====================
125
126    return e(t, x, q, spline_c, spline_l, spline_r) + r_c(u) + r_w(u)

129def generate_D(tau: np.ndarray) -> np.ndarray:
130    """
131    Generates differentiation matrix using Barycentric weights
132
133    Args:
134        tau (np.ndarray): 1D numpy array containing LG points and -1 and 1
135
136    Returns:
137        D (np.ndarray): Differentiation matrix
138    """
139    D = np.zeros((len(tau), len(tau)))
140    w = np.zeros(len(tau))
141    # Precomputes Barycentric weights (denom) for fp/numeric stability
142
143    for j in range(len(tau)):
144        p = 1.0
145        for i in range(len(tau)):
146            if i != j:
147                p *= tau[j] - tau[i]
148        w[j] = 1.0 / p
149
150    for i in range(len(tau)):
151        for j in range(len(tau)):
152            if i != j:
153                D[i, j] = w[j] / w[i] / (tau[i] - tau[j])
154        D[i, i] = -np.sum(D[i, :])
155    return D

158def fit_iteration(
159    t: np.ndarray,
160    N: np.ndarray,
161    spline_c: BSpline,
162    spline_l: BSpline,
163    spline_r: BSpline,
164):
165    """
166    Runs a single iteration of pseudospectral collocation.
167
168    Args:
169        t (np.ndarray): 1D numpy array containing the mesh points
170        N (np.ndarray): 1D numpy array containing the number of collocation points in each interval
171
172
173    Returns:
174        tbd
175    """
176    SAMPLING_RES = 5
177
178    # Normalization stage
179
180    t_fine = np.linspace(t[0], t[-1], t[-1] // SAMPLING_RES)
181    x_min, y_min, z_min = splev(t[0], spline_c)
182    x_max, y_max, z_max = x_min, y_min, z_min
183    
184    c_sample = splev(t_fine, spline_c)
185    l_sample = splev(t_fine, spline_l)
186    r_sample = splev(t_fine, spline_r)
187
188    
189    
190    
191    # for 
192
193    # Scaling arc length
194    t_final = t[-1]
195    t = t / t_final
196
197    opti = Opti()
198
199    K = len(N)  # Number of time intervals
200
201    # State and configuration + derivative matrices. For any matrix M that represents a function
202    # m(tau) on a segment k, M_ij = m_j(tau_i)
203
204    # Q, dQ, ddQ are (N_k + 2) x (n_q).
205    Q = []  # Array containing Q matrices. q_j = [theta, mu, phi, n_l, n_r].
206    dQ = []  # Array containing dQ (1st der) matrices.
207    ddQ = []  # Array containing ddQ (2nd der) matrices.
208
209    # X, dX are (N_k + 2) x (n_x).
210    X = []  # Array containing X matrices. x_j = [x,y,z].
211    dX = []  # Array containing dX (1st der matrices).
212
213    J = 0  # Cost accumulator
214
215    # Stores the theta values for the initial guess
216    theta_arr = []
217
218    # Constraints for each segment k
219    for k in range(K):
220        # Useful values for conversion between t and tau
221        norm_factor = (t[k + 1] - t[k]) / 2
222        t_tau_0 = (t[k + 1] + t[k]) / 2  # Global time t at tau = 0
223
224        # Legendre Gauss collocation points tau with appended -1 and 1
225        # w is the quadrature weights
226        tau, w = np.polynomial.legendre.leggauss(N[k])
227        tau = np.asarray([-1] + list(tau) + [1])
228
229        # Global time (t) at collocation points
230        t_tau = norm_factor * tau + t_tau_0
231
232        # Differentiation matrix
233        D = generate_D(tau)
234
235        # Generates X and Q matrices + derivatives for this segment
236        if k == 0:
237            Q.append(opti.variable(N[k] + 2, n_q))
238            X.append(opti.variable(N[k] + 2, n_x))
239        else:
240            # Explicitly couples last of previous segment with first of current segment
241            # by setting them as the same variable
242            Q.append(vertcat(Q[k - 1][-1, :], opti.variable(N[k] + 1, n_q)))
243            X.append(vertcat(X[k - 1][-1, :], opti.variable(N[k] + 1, n_x)))
244
245        dQ.append((2 / (t[k + 1] - t[k])) * mtimes(D, Q[k]))
246        ddQ.append(2 / (t[k + 1] - t[k]) * mtimes(D, dQ[k]))
247        dX.append((2 / (t[k + 1] - t[k])) * mtimes(D, X[k]))
248
249        # Continuity constraints
250        if k != 0:
251            opti.subject_to(dQ[k - 1][-1, :] == dQ[k][0, :])
252            opti.subject_to(ddQ[k - 1][-1, :] == ddQ[k][0, :])
253
254        # Collocation constraints (enforces dynamics on X)
255        theta = Q[k][:-1, 0]
256        mu = Q[k][:-1, 1]
257
258        opti.subject_to(dX[k][:-1, 0] == cos(theta) * cos(mu))
259        opti.subject_to(dX[k][:-1, 1] == sin(theta) * cos(mu))
260        opti.subject_to(dX[k][:-1, 2] == -sin(mu))
261        opti.subject_to(opti.bounded(-pi / 2 + 1e-3, mu, pi / 2 - 1e-3))
262
263        # Quadrature enforcement
264        for j in range(N[k]):
265
266            lagrange_term = g(
267                t_tau[j + 1],
268                X[k][j + 1, :],
269                Q[k][j + 1, :],
270                ddQ[k][j + 1, :],
271                spline_c,
272                spline_l,
273                spline_r,
274            )
275
276            J += norm_factor * w[j] * lagrange_term
277
278        # Note: by continuity constraints it is guaranteed X[k + 1][0, :] == X[k][-1, :]
279        # opti.subject_to(defect == X[k][-1, :])
280
281        # ========================== Generates initial guesses for variables  ==========================
282
283        # Guesses X values based on spline position
284        opti.set_initial(X[k], np.asarray(splev(t_tau, spline_c)).T)
285
286        # Creates tangent and normal vectors to find Euclidean angles theta, mu, phi
287        # tangent estimate = (spline_c)' / ||t||
288        tangent = np.asarray(splev(t_tau, spline_c, der=1)).T
289        tangent = tangent / np.linalg.norm(tangent, axis=1)[:, np.newaxis]
290
291        # normal estimate = (spline_l - spline_c) / ||n||
292        normal = (
293            np.asarray(splev(t_tau, spline_l)) - np.asarray(splev(t_tau, spline_c))
294        ).T
295        normal = normal / np.linalg.norm(normal, axis=1)[:, np.newaxis]
296
297        # Calculates Euclidean angles
298        mu_guess = np.asin(-tangent[:, 2])
299        theta_guess = np.arctan2(tangent[:, 1], tangent[:, 0])
300        phi_guess = np.asin(normal[:, 2] / np.cos(mu_guess))
301        theta_arr.append(theta_guess)
302
303        opti.set_initial(Q[k][:, 1], mu_guess)
304        opti.set_initial(Q[k][:, 2], phi_guess)
305
306        # Estimates left-right boundary lengths
307        nl_guess = np.linalg.norm(
308            (np.asarray(splev(t_tau, spline_l)) - np.asarray(splev(t_tau, spline_c))).T,
309            axis=1,
310        )
311        nr_guess = -np.linalg.norm(
312            (np.asarray(splev(t_tau, spline_r)) - np.asarray(splev(t_tau, spline_c))).T,
313            axis=1,
314        )
315
316        opti.set_initial(Q[k][:, 3], nl_guess)
317        opti.set_initial(Q[k][:, 4], nr_guess)
318
319    # Unwrapping / removing theta jump discontinuity
320    # TODO there are many more efficient ways to do this
321    last = theta_arr[0][0]
322    for k, seg in enumerate(theta_arr):
323        for i, theta in enumerate(seg):
324            diff = theta - last
325            while diff > np.pi or diff < -np.pi:
326                if diff > np.pi:
327                    seg[i] -= 2 * np.pi
328                elif diff < -np.pi:
329                    seg[i] += 2 * np.pi
330                diff = seg[i] - last
331            last = seg[i]
332        opti.set_initial(Q[k][:, 0], seg)
333    #     print(seg[0])
334    # print(theta_arr[-1][-1])
335
336    # Initial conditions
337    x0 = splev(0, spline_c)
338    for i in range(3):
339        opti.subject_to(X[0][0, i] == x0[i])
340
341    # Periodicity
342    opti.subject_to(X[-1][-1, :] == X[0][0, :])
343
344    # opti.subject_to(sin(Q[-1][-1, 0]) == sin(Q[0][0, 0]))
345    # opti.subject_to(cos(Q[-1][-1, 0]) == cos(Q[0][0, 0]))
346    opti.subject_to(Q[-1][-1, 0] == Q[0][0, 0] - 2 * pi)
347    opti.subject_to(Q[-1][-1, 1:] == Q[0][0, 1:])
348    opti.subject_to(dQ[-1][-1, :] == dQ[0][0, :])
349    opti.subject_to(ddQ[-1][-1, :] == ddQ[0][0, :])
350
351    # Optimize!
352    solver_options = {
353        "ipopt.print_level": 5,
354        "print_time": 0,
355        "ipopt.sb": "no",
356        "ipopt.max_iter": 1000,
357        "detect_simple_bounds": True,
358        "ipopt.mu_strategy": "adaptive",
359        "ipopt.nlp_scaling_method": "gradient-based",
360        "ipopt.bound_relax_factor": 1e-8,
361        "ipopt.honor_original_bounds": "yes",
362    }
363
364    opti.minimize(J)
365    opti.solver("ipopt", solver_options)
366    try:
367        sol = opti.solve()
368    except:
369        sol = opti.debug
370
371    # Collect solution
372    # Matrices
373    solution_x = []
374    solution_q = []
375
376    # Per interval
377    X_sol = []
378    Q_sol = []
379
380    for k, segment_q in enumerate(Q):
381        segment_q = sol.value(segment_q)
382        Q_sol.append(segment_q)
383        for i in range(N[k] + 2):
384            solution_q.append(segment_q[i, :])
385
386    for k, segment_x in enumerate(X):
387        segment_x = sol.value(segment_x)
388        X_sol.append(segment_x)
389        for i in range(N[k] + 2):
390            solution_x.append(segment_x[i, :])
391
392    return np.asarray(solution_x), np.asarray(solution_q), X_sol, Q_sol

395def fit_track(
396    spline_c: BSpline,
397    spline_l: BSpline,
398    spline_r: BSpline,
399    max_dist: float,
400    refinement_steps: int = 3,
401) -> Track:
402    """
403    Fits a Track object
404
405    Args:
406        spline_c (BSpline): Scipy BSpline for center line
407        spline_l (BSpline): Scipy BSpline for left boundary
408        spline_r (BSpline): Scipy BSpline for right boundary
409        max_dist (float): Length of track
410        refinement_steps (int, optional): Number of mesh refinement steps. Defaults to 4.
411
412    Returns:
413        Track: _description_
414    """
415    INITIAL_COLLOCATION = 5
416    INITIAL_POINTS = 10
417
418    t = np.linspace(0, max_dist, INITIAL_POINTS)  # Mesh points
419    N = np.array(
420        [INITIAL_COLLOCATION] * (INITIAL_POINTS - 1)
421    )  # Collocation points per interval
422
423    # Initial track
424    X, Q, X_mat, Q_mat = fit_iteration(t, N, spline_c, spline_l, spline_r)
425    track = Track(Q_mat, X_mat, t)
426
427    # Refinement
428    for i in range(refinement_steps):
429        print(f"Refinement step {i + 1}/{refinement_steps}")
430        N, t = mesh_refinement_iteration(track, t, N, spline_c, spline_l, spline_r)
431        X, Q, X_mat, Q_mat = fit_iteration(t, N, spline_c, spline_l, spline_r)
432        track = Track(Q_mat, X_mat, t)
433
434    return track

437def mesh_refinement_iteration(
438    track: Track,
439    t: np.ndarray,
440    N: np.ndarray,
441    spline_c: BSpline,
442    spline_l: BSpline,
443    spline_r: BSpline,
444    resolution=0.5,
445    variation_thres=0.7,
446    divides=3,
447    degree_increase=5,
448    initial_points=10,
449) -> tuple[np.ndarray, np.ndarray]:
450    """
451    Performs a single iteration of mesh refinement
452
453    Args:
454        track (Track): Track object
455        t (np.ndarray): Array of arc length parameters giving mesh points
456        N (np.ndarray): Array containing the number of collocation points per
457                        interval (length = len(t) - l)
458        spline_c (BSpline): Scipy BSpline for center line
459        spline_l (BSpline): Scipy BSpline for left boundary
460        spline_r (BSpline): Scipy BSpline for right boundary
461        resolution (float, optional): Sampling resolution. Defaults to 0.5.
462        variation_thres (float, optional): Threshold on ratio of stdev to mean. Defaults to 0.4.
463        divides (int, optional): Number of mesh points added when dividing. Defaults to 2.
464        degree_increase (int, optional): Degree increase when adding collocation points. Defaults to 5.
465        initial_points (int, optional): Starting number of collocation points when
466                                        a new interval is formed. Defaults to 3.
467
468    Returns:
469        tuple[np.ndarray, np.ndarray]: Refined N and t
470    """
471    print(N.shape, t.shape)
472
473    # Start t of each interval
474    interval_starts = track.t[:-1]
475    new_t, new_N = [t.min()], []
476    interval_costs = []
477    geo_mean_cost = 0
478
479    deg = 0
480    div = 0
481    skip = 0
482
483    for i, start_t in enumerate(interval_starts):
484        end_t = track.t[i + 1]
485
486        # Sample t, remove first so it cannot be added multiple times
487        # assert end_t != start_t
488        sample_t = np.linspace(
489            start_t, end_t, math.ceil((end_t - start_t) / resolution)
490        )[1:]
491
492        # Sample state at each t as well as its derivative
493        states = track.state(sample_t)
494        control = track.der_state(sample_t, der=2)
495
496        # Compute costs across interval i at the end of each t
497        costs = np.asarray(
498            [
499                g(
500                    sample_t[j],
501                    state[:3],
502                    state[3:],
503                    control[j][3:],
504                    spline_c,
505                    spline_l,
506                    spline_r,
507                )
508                for j, state in enumerate(states)
509            ]
510        )
511
512        geo_mean_cost += np.log(costs.mean())
513
514        interval_costs.append(costs)
515
516    geo_mean_cost = np.exp(geo_mean_cost / len(interval_costs))
517
518    for i, start_t in enumerate(interval_starts):
519        end_t = track.t[i + 1]
520        sample_t = np.linspace(
521            start_t, end_t, math.ceil((end_t - start_t) / resolution)
522        )[1:]
523
524        costs = interval_costs[i]
525        
526        stdev = costs.std()
527        mean = costs.mean()
528
529        print(mean, geo_mean_cost)
530
531        if mean < geo_mean_cost:
532            new_N.append(N[i])
533            new_t.append(end_t)
534            print("interval is good")
535            skip += 1
536            continue
537
538        print("variation", stdev / mean)
539        total = np.sum(costs)
540
541        if stdev / mean > variation_thres:
542            # Divide
543            div += 1
544            cumulative = 0
545
546
547            for j, c in enumerate(costs):
548                cumulative += c
549                if cumulative > total / divides:
550                    cumulative = 0
551
552                    new_N.append(initial_points)
553                    new_t.append(sample_t[j])
554
555            if abs(end_t - new_t[-1]) > 1e-7:
556                new_N.append(initial_points)
557                new_t.append(end_t)
558
559        else:
560            # Increase degree
561            deg += 1
562            new_N.append(N[i] + degree_increase)
563
564            new_t.append(end_t)
565
566    print(f"Degree increased: {deg}\tDivided: {div}\tSkipped: {skip}")
567    print(np.asarray(new_N).shape, np.asarray(new_t).shape)
568    assert len(new_N) + 1 == len(new_t)
569    return np.asarray(new_N), np.asarray(new_t)

 14    """
 15    Calculates path error g(t) at a specific t.
 16    For track fitting, used both for computing the symbolic cost in CasADi
 17    for optimization and evaluating the numeric generated cost for
 18    hp-refinement.
 19    g(t)
 20
 21    Args:
 22        t (float): Arc length parameter
 23        x (CasADi Expression | np.ndarray): Vector containing [x, y, z]
 24        q (CasADi Expression | np.ndarray): Vector containing [θ, μ, Φ, n_l, n_r]
 25        u (CasADi Expression | np.ndarray): The control vector of second derivatives. u = ddq
 26        spline_c (BSpline): Scipy BSpline for center line
 27        spline_l (BSpline): Scipy BSpline for left boundary
 28        spline_r (BSpline): Scipy BSpline for right boundary
 29
 30    Returns:
 31        CasADi Expression | float: g(.)
 32    """
 33
 34    # ==================== Start Defining sub-error functions ====================
 35    def e(
 36        t: float,
 37        x,
 38        q,
 39        spline_c: BSpline,
 40        spline_l: BSpline,
 41        spline_r: BSpline,
 42        w_c: float = 1e-3,
 43        w_l: float = 1e-3,
 44        w_r: float = 1e-3,
 45    ):
 46        """
 47        Computes tracking error
 48        e = w_c ||x - c_spline(t)||^2 + w_l ||b_l - l_spline(t)||^2 + w_r ||b_r - r_spline(t)||^2
 49        where
 50        b_l = x + n * n_l
 51        b_r = x + n * n_r
 52        n = second column of Euler Rotation Matrix
 53
 54        Args:
 55            t (float): Arc length parameter
 56            x (CasADi Expression | np.ndarray): Vector containing [x, y, z]
 57            q (CasADi Expression | np.ndarray): Vector containing [theta, mu, phi]
 58            spline_c (BSpline): Scipy BSpline for center line
 59            spline_l (BSpline): Scipy BSpline for left boundary
 60            spline_r (BSpline): Scipy BSpline for right boundary
 61            w_c (float, optional): Defaults to 1e-3.
 62            w_l (float, optional): Defaults to 1e-3.
 63            w_r (float, optional): Defaults to 1e-3.
 64
 65        Returns:
 66            CasADi Expression | np.ndarray: Tracking error
 67        """
 68
 69        theta = q[0]
 70        mu = q[1]
 71        phi = q[2]
 72        n_l = q[3]
 73        n_r = q[4]
 74
 75        n = vertcat(
 76            cos(theta) * sin(mu) * sin(phi) - sin(theta) * cos(phi),
 77            sin(theta) * sin(mu) * sin(phi) + cos(theta) * cos(phi),
 78            cos(mu) * sin(phi),
 79        )
 80
 81        b_l = x.T + n * n_l
 82        b_r = x.T + n * n_r
 83
 84        x_c, y_c, z_c = splev(t, spline_c)
 85        x_l, y_l, z_l = splev(t, spline_l)
 86        x_r, y_r, z_r = splev(t, spline_r)
 87
 88        return (
 89            w_c * ((x[0] - x_c) ** 2 + (x[1] - y_c) ** 2 + (x[2] - z_c) ** 2)
 90            + w_l * ((b_l[0] - x_l) ** 2 + (b_l[1] - y_l) ** 2 + (b_l[2] - z_l) ** 2)
 91            + w_r * ((b_r[0] - x_r) ** 2 + (b_r[1] - y_r) ** 2 + (b_r[2] - z_r) ** 2)
 92        )
 93
 94    def r_c(u, w_theta=3e3, w_mu=1e9, w_phi=1e9):
 95        """
 96        Computes the error term that penalizes track curvature
 97        r_c = w_theta * dd_theta^2 + w_mu * dd_mu^2 + w_phi * dd_phi^2
 98
 99        Args:
100            u (CasADi Expression | np.ndarray): The control vector of second derivatives. u = ddq
101            w_theta (float, optional): Defaults to 3e3.
102            w_mu (float, optional): Defaults to 1e9.
103            w_phi (float, optional): Defaults to 1e9.
104
105        Returns:
106            CasADi Expression | np.ndarray: r_c
107        """
108        return w_theta * u[0] ** 2 + w_mu * u[1] ** 2 + w_phi * u[2] ** 2
109
110    def r_w(u, w_n_l=1e5, w_n_r=1e5):
111        """
112        Computes the error term that penalizes track boundary noise
113        r_w = w_n_l * dd_n_l^2 + w_n_r * dd_n_r^2
114
115        Args:
116            u (CasADi Expression | np.ndarray): The control vector of second derivatives. u = ddq
117            w_n_l (_type_, optional): Defaults to 1e2.
118            w_n_r (_type_, optional): Defaults to 1e2.
119
120        Returns:
121            CasADi Expression | np.ndarray: r_w
122        """
123        return w_n_l * u[3] ** 2 + w_n_r * u[4] ** 2
124
125    # ==================== End Defining sub-error functions ====================
126
127    return e(t, x, q, spline_c, spline_l, spline_r) + r_c(u) + r_w(u)

130def generate_D(tau: np.ndarray) -> np.ndarray:
131    """
132    Generates differentiation matrix using Barycentric weights
133
134    Args:
135        tau (np.ndarray): 1D numpy array containing LG points and -1 and 1
136
137    Returns:
138        D (np.ndarray): Differentiation matrix
139    """
140    D = np.zeros((len(tau), len(tau)))
141    w = np.zeros(len(tau))
142    # Precomputes Barycentric weights (denom) for fp/numeric stability
143
144    for j in range(len(tau)):
145        p = 1.0
146        for i in range(len(tau)):
147            if i != j:
148                p *= tau[j] - tau[i]
149        w[j] = 1.0 / p
150
151    for i in range(len(tau)):
152        for j in range(len(tau)):
153            if i != j:
154                D[i, j] = w[j] / w[i] / (tau[i] - tau[j])
155        D[i, i] = -np.sum(D[i, :])
156    return D

Arguments:
  • track (Track): Track object
  • t (np.ndarray): Array of arc length parameters giving mesh points

  • initial_points (int, optional): Starting number of collocation points when a new interval is formed. Defaults to 3.

tuple[np.ndarray, np.ndarray]: Refined N and t
